public class Horner {

	public static void main(String[] args) {
		Horner h = new Horner();
		int[] a = { 1, 3, 5 };
		System.out.println("a(1)=" + h.horner(a, a.length - 1, 1));
		System.out.println("a(2)=" + h.horner(a, a.length - 1, 2));
	}

	int horner(int[] a, int n, int x) {
		int result = a[n];
		for (int i = n - 1; i >= 0; --i) {
			result = result * x + a[i];
			/**/System.out.println("i=" + i + " result" + result);
		}
		return result;
	}
}
